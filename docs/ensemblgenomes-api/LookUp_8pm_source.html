00038 =head1 DESCRIPTION
00039 
00040 This module is a helper that provides additional methods to aid navigating a registry of >6000 species across >25 databases. 
00041 It does not replace the Registry but provides some additional methods for finding species e.g. by searching for species that 
00042 have an alias that match a regular expression, or species which are derived from a specific ENA/INSDC accession, or species
00043 that belong to a particular part of the bacterial taxonomy. 
00044 
00045 There are a number of ways of creating a lookup. The simplest and fastest is to supply a URL for a JSON resource that contains all the details
00046 needed to connect to the public Ensembl Genomes bacterial databases e.g.
00047 
00048     my $lookup = Bio::EnsEMBL::LookUp->new(-URL=>'http://bacteria.ensembl.org/registry.json');
00049 
00050 Alternatively, a local file containing the required JSON can be specified instead:
00051 
00052     my $lookup = Bio::EnsEMBL::LookUp->new(-FILE=>"/path/to/reg.json");
00053 
00054 Finally, a Registry already loaded with the ENA core databases can be supplied:
00055 
00056     my $lookup = Bio::EnsEMBL::LookUp->new(-REGISTRY=>'Bio::EnsEMBL::Registry');
00057 
00058 If the standard Registry is used, the argument can be omitted completely:
00059 
00060     my $lookup = Bio::EnsEMBL::LookUp->new();
00061 
00062 To populate the registry with just the Ensembl Bacteria databases for the current software release on a specified server, the following method can be used:
00063 
00064     Bio::EnsEMBL::LookUp->register_all_dbs( $host,
00065        $port, $user, $pass);
00066 
00067 Once a lookup has been created, there are various methods to retreive DBAdaptors for species of interest:
00068 
00069 1. To find species by name - all DBAdaptors for species with a name or alias matching the supplied string:
00070 
00071     $dbas = $lookup->get_by_name_exact('Escherichia coli str. K-12 substr. MG1655');
00072 
00073 2. To find species by name pattern - all DBAdaptors for species with a name or alias matching the supplied regexp:
00074 
00075     $dbas = $lookup->get_by_name_exact('Escherichia coli .*);
00076 
00077 3. To find species with the supplied taxonomy ID:
00078 
00079     $dbas = $lookup->get_all_by_taxon_id(388919);
00080     
00081 4. In coordination with a taxonomy node adaptor to get DBAs for all descendants of a node:
00082 
00083     my $node = $node_adaptor->fetch_by_taxon_id(511145);
00084     for my $child (@{$node_adaptor->fetch_descendants($node)}) {
00085         my $dbas = $lookup->get_all_by_taxon_id($node->taxon_id())
00086         if(defined $dbas) {
00087             for my $dba (@{$dbas}) {
00088                 # do something with the $dba
00089             }
00090         }
00091     }
00092 
00093 The retrieved DBAdaptors can then be used as normal e.g.
00094 
00095     for my $gene (@{$dba->get_GeneAdaptor()->fetch_all_by_biotype('protein_coding')}) {
00096         print $gene->external_name."\n";
00097     }
00098 
00099 Once retrieved, the arguments needed for constructing a DBAdaptor directly can be dumped for later use e.g.
00100 
00101     my $args = $lookup->dba_to_args($dba);
00102     ... store and retrieve $args for use in another script ... 
00103     my $resurrected_dba = Bio::EnsEMBL::DBSQL::DBAdaptor->new(@$args);

00378   my ($self) = @_;

00380   my $dbc_hash;
00381   my $dba_hash;
00382   for my $dba (@{$self->{dbas}}) {
00383     my $dbc_loc = _dbc_to_locator($dba->dbc());
00384     $dbc_hash->{$dbc_loc} = $dba->dbc();
00385     push @{$dba_hash->{$dbc_loc}}, $dba;
00386   }
00387   # create auxillary hashes

00389   my $name_hash  = _invert_dba_hash($self->{dbas_by_name});
00390   my $taxid_hash = _invert_dba_hash($self->{dbas_by_taxid});
00391   my $gc_hash    = _invert_dba_hash($self->{dbas_by_vgc});
00392   # create array of hashes
00393   my $out_arr;
00394   while (my ($dbc_loc, $dbconn) = each(%{$dbc_hash})) {
00395     my $dbc_h = {driver   => $dbconn->driver(),
00396                  host     => $dbconn->host(),
00397                  port     => $dbconn->port(),
00398                  username => $dbconn->username(),
00399                  password => $dbconn->password(),
00400                  dbname   => $dbconn->dbname(),
00401                  species  => []};
00402     for my $dba (@{$dba_hash->{$dbc_loc}}) {
00403       my $dba_loc = _dba_to_locator($dba);
00404       my $gc      = $gc_hash->{$dba_loc};
